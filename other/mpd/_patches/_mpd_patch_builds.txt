

/////////////////////////////////////////////
//
// BUILD MPD WITH PATCHES
//
/////////////////////////////////////////////

sudo ./dev.sh mount-moode

# Patch availability bitmask
SELECTIVE_RESAMPLING=1
SOX_CUSTOM_RECIPE=2

# Patch identifier
PATCH_ID=_p0x$(($SELECTIVE_RESAMPLING + $SOX_CUSTOM_RECIPE))

# Ex: mpd-0.21.24_p0x3
# Ex: mpd-0.22~git_p0x3

# Sources
# Release version
sudo cp /mnt/moode-player/GitHub/moode/other/mpd/mpd-0.21.24.tar.xz ./
sudo tar xf ./mpd-0.21.24.tar.xz
# Git master version
sudo cp /mnt/moode-player/GitHub/moode/other/mpd/mpd-0.22~git-20200723-dbdf782.zip ./
sudo unzip -q ./mpd-0.22~git-20200723-dbdf782.zip

# Patch files
sudo cp /mnt/moode-player/GitHub/moode/other/mpd/_patches/*.patch ./

# Patch the sources
VERSION=21
cd ~/mpd-0.21.24
sudo patch -p1 < ~/mpd-0.$VERSION.x_soxr_customrecipe.patch
sudo patch -p1 < ~/mpd_0.$VERSION.xx_selective_resample_mode.patch
sudo cp meson.build meson.build.orig
sudo sed -i "/version: '0/s/',/$PATCH_ID',/" ./meson.build
VERSION=22
cd ~/MPD-master
sudo patch -p1 < ~/mpd-0.$VERSION.x_soxr_customrecipe.patch
sudo patch -p1 < ~/mpd_0.$VERSION.xx_selective_resample_mode.patch
sudo cp meson.build meson.build.orig
sudo sed -i "/version: '0/s/',/$PATCH_ID',/" ./meson.build

# Compile MPD 0.21.y
cd ~/mpd-0.21.24
sudo meson . output/release --buildtype=release -Db_ndebug=true
sudo meson configure -Dalsa=enabled -Dbzip2=enabled -Dcurl=enabled -Ddatabase=true -Ddsd=true -Dffmpeg=enabled -Dfaad=enabled -Dflac=enabled -Dhttpd=true -Did3tag=enabled -Dlame=enabled -Dlibmpdclient=enabled -Dmad=enabled -Dmpg123=enabled -Dpipe=true -Drecorder=true -Dshout=enabled -Dsoundcloud=enabled -Dsoxr=enabled -Dvorbis=enabled -Dwave_encoder=true -Dwavpack=enabled -Dzeroconf=avahi -Dzzip=enabled -Dao=disabled -Daudiofile=disabled -Ddbus=disabled -Dexpat=disabled -Dfluidsynth=disabled -Dgme=disabled -Dipv6=disabled -Djack=disabled -Dlibsamplerate=disabled -Dnfs=disabled -Doss=disabled -Dpulse=disabled -Dsidplay=disabled -Dsmbclient=disabled -Dsndfile=disabled -Dsqlite=disabled -Dudisks=disabled -Dupnp=disabled -Dwildmidi=disabled output/release
sudo ninja -C output/release
# Copy binary
sudo cp output/release/mpd /mnt/SDCARD/mpd-0.21.24$PATCH_ID
sudo cp output/release/mpd /var/local/www/mpd_versions/mpd-0.21.24$PATCH_ID

# Compile MPD 0.22~git
cd ~/MPD-master
sudo meson . output/release --buildtype=release -Db_ndebug=true
sudo meson configure -Dalsa=enabled -Dbzip2=enabled -Dcurl=enabled -Ddatabase=true -Ddsd=true -Dffmpeg=enabled -Dfaad=enabled -Dflac=enabled -Dhttpd=true -Did3tag=enabled -Dlame=enabled -Dlibmpdclient=enabled -Dmad=enabled -Dmpg123=enabled -Dpipe=true -Drecorder=true -Dshout=enabled -Dsoundcloud=enabled -Dsoxr=enabled -Dvorbis=enabled -Dwave_encoder=true -Dwavpack=enabled -Dzeroconf=avahi -Dzzip=enabled -Dao=disabled -Daudiofile=disabled -Ddbus=disabled -Dexpat=disabled -Dfluidsynth=disabled -Dgme=disabled -Dipv6=disabled -Djack=disabled -Dlibsamplerate=disabled -Dnfs=disabled -Doss=disabled -Dpulse=disabled -Dsidplay=disabled -Dsmbclient=disabled -Dsndfile=disabled -Dsqlite=disabled -Dudisks=disabled -Dupnp=disabled -Dwildmidi=disabled output/release
sudo ninja -C output/release
# Copy binary
sudo cp output/release/mpd /mnt/SDCARD/mpd-0.22~git$PATCH_ID
sudo cp output/release/mpd /var/local/www/mpd_versions/mpd-0.22~git$PATCH_ID

cd ~
sudo rm -rf ./mpd*
sudo rm -rf ./MPD*
